    def punctuate!(string : String) : String #

    Adds a period to the end of a string if no terminating punctuation (?, !, .) is present.

    Myutils.punctuate!("let's end this")  # => "let's end this."
Myutils.punctuate!("let's end this!") # => "let's end this!"
Myutils.punctuate!("let's end this?") # => "let's end this?"
